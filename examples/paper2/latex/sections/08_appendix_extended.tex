\begin{algorithm}[h]
\caption{COCO ROI crop (union + padding)}
\begin{algorithmic}[1]
\STATE \textbf{Input:} image $I$, boxes $\{b_j=[x,y,w,h]\}$, padding $p$
\STATE $B \leftarrow \mathrm{union}(\{b_j\})$
\STATE Expand $B$ by fraction $p$ on each side; clamp to image bounds
\STATE \RETURN $\mathrm{crop}(I, B)$
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[h]
\caption{B3 inference (MobileNetV3-Small)}
\begin{algorithmic}[1]
\STATE \textbf{Input:} RGB image, optional boxes
\STATE $I' \leftarrow \mathrm{ROIcrop}(I,\mathrm{boxes})$
\STATE $x \leftarrow \mathrm{normalize}(\mathrm{resize}(I', 224))$
\STATE $p \leftarrow \mathrm{softmax}(\mathrm{MobileNetV3}(x))$
\STATE \RETURN $Q_{\mathrm{img}} \leftarrow p_{\mathrm{clear}}$
\end{algorithmic}
\end{algorithm}
