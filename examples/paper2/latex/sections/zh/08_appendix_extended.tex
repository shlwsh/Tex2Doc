\begin{algorithm}[t]
\caption{COCO ROI 裁剪（并集 + padding）}
\begin{algorithmic}[1]
\STATE \textbf{输入：} 图像 $I$，框集合 $\{b_j=[x,y,w,h]\}$，padding $p$
\STATE $B \leftarrow$ 所有框的并集
\STATE 各边按 $p$ 比例外扩并裁剪至图像边界
\RETURN $\mathrm{crop}(I, B)$
\end{algorithmic}
\end{algorithm}
