\begin{abstract}
We report the first experimental observation of long-range entanglement
in a topological insulator heterostructure at room temperature. Our measurements
reveal sustained quantum coherence across 100 nanometers, exceeding previous
records by an order of magnitude. This finding has profound implications for
fault-tolerant quantum computing and topological quantum information processing.
\end{abstract}

\section{INTRODUCTION}

Topological phases of matter have revolutionized our understanding of
quantum systems~\cite{qi2011topological,hasan2010topological}.
The discovery of topological insulators revealed a new class of materials
with insulating bulk and conducting surface states protected by time-reversal symmetry.

Quantum entanglement lies at the heart of quantum information science.
However, maintaining entanglement over macroscopic distances remains a grand
challenge~\cite{preskill1998quantum,bennett1988concentrating}.

\section{THEORETICAL FRAMEWORK}

The entanglement entropy in a topological phase follows the area law:
\begin{equation}
S \sim \frac{c}{12}\log L + \gamma + O(1/L) \label{eq:area_law}
\end{equation}

The master equation governing the density matrix evolution is:
\begin{equation}
\begin{gathered}
\rho(t) = \rho_0 - \frac{i}{\hbar}[H,\rho_0]t
          -\frac{1}{2\hbar^2}[H,[H,\rho_0]]t^2
          +\cdots \label{eq:master}
\end{gathered}
\end{equation}

\section{LINDBLAD EQUATION}

For open quantum systems, the Lindblad equation describes dissipation:
\begin{equation}
\frac{d\rho}{dt} = -\frac{i}{\hbar}[H,\rho] + \sum_{k}\Bigl(
L_k\rho L_k^\dagger - \tfrac{1}{2}\{L_k^\dagger L_k,\rho\}\Bigr) \label{eq:lindblad}
\end{equation}

The von Neumann entropy is:
\begin{equation}
\begin{multline}
S(\rho_{AB}) = -\tr(\rho_{AB}\log_2\rho_{AB})
              = -\tr_A\tr_B\bigl(\rho_{AB}(\log_2\rho_{AB})\bigr) \\
              \le -\tr_A\tr_B\bigl(\rho_{AB}(\log_2\rho_A\otimes\rho_B)\bigr)
              = S(\rho_A) + S(\rho_B) .
\end{multline} \label{eq:entropy}
\end{equation}
